001package org.anarres.linguistics;
002
003import java.util.Locale;
004import java.util.Map;
005
006import org.anarres.linguistics.core.*;
007import org.anarres.linguistics.en.*;
008
009public class Registry {
010
011    private static class Key<T> {
012
013        private final Locale locale;
014        private final Class<T> iface;
015
016        public Key(Locale locale, Class<T> iface) {
017            this.locale = locale;
018            this.iface = iface;
019        }
020
021        @Override
022        public int hashCode() {
023            return locale.hashCode() ^ iface.hashCode();
024        }
025
026        @Override
027        public boolean equals(Object o) {
028            if (this == o)
029                return true;
030            if (!(o instanceof Key))
031                return false;
032            Key k = (Key) o;
033            return locale.equals(k.locale)
034                    && iface.equals(k.iface);
035        }
036    }
037
038    private static Map<Key<Object>, Object> handlers;
039
040    static {
041        add(Locale.UK, Pluralizer.class, new EnglishPluralizer(Locale.UK));
042        add(Locale.US, Pluralizer.class, new EnglishPluralizer(Locale.US));
043    }
044
045    public static <T> T get(Locale locale, Class<T> iface) {
046        Object obj = handlers.get(new Key<T>(locale, iface));
047        if (obj == null)
048            throw new IllegalArgumentException(
049                    "No handler available for " + locale + " and " + iface
050            );
051        return (T) obj;
052    }
053
054    public static <T> void add(Locale locale, Class<T> iface, T obj) {
055        handlers.put(new Key(locale, iface), obj);
056    }
057
058}
