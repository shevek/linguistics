001package org.anarres.linguistics.core;
002
003import java.util.Locale;
004
005/**
006 * The base class of all linguistics handlers.
007 */
008public abstract class Base {
009
010    private final Locale locale;
011
012    protected Base(Locale locale) {
013        this.locale = locale;
014    }
015
016    /**
017     * Returns the locale of this handler.
018     */
019    public Locale getLocale() {
020        return locale;
021    }
022
023    /** Returns 'word' with the last 'trim' characters removed. */
024    protected String prefix(String word, int trim) {
025        return word.substring(0, word.length() - trim);
026    }
027
028    /** Returns the last 'length' characters of 'word'. */
029    protected String suffix(String word, int length) {
030        return word.substring(word.length() - length);
031    }
032
033}
