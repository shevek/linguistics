001package org.anarres.linguistics.core;
002
003import java.util.Arrays;
004import java.util.Collections;
005import java.util.List;
006import java.util.Locale;
007
008/**
009 * A conjugation routine for verbs.
010 */
011public abstract class Conjugator<E extends Enum> extends Base {
012
013    private List<E> conj;
014
015    protected Conjugator(Locale locale) {
016        super(locale);
017    }
018
019    /**
020     * Sets the list of allowed conjugations (people).
021     *
022     * This is usually called from a subclass constructor as:
023     * <code>setConjugation(MyConjugation.values());</code>
024     */
025    protected void setConjugation(E[] values) {
026        this.conj = Collections.unmodifiableList(Arrays.asList(values));
027    }
028
029    /**
030     * Returns the list of allowed conjugations (people).
031     *
032     * These will be the values of an Enum.
033     */
034    public List<E> getConjugations() {
035        return conj;
036    }
037
038    /**
039     * Conjugates a verb for the given person.
040     *
041     * @param verb The root form of the verb.
042     * @param person The person with respect to whom the verb is to be conjugated.
043     */
044    public abstract String getConjugation(String verb, E person);
045}
