001package org.anarres.linguistics.core;
002
003import java.util.Locale;
004import javax.annotation.Nonnull;
005
006/**
007 * A pluralizer for nouns.
008 */
009public abstract class Pluralizer extends Base {
010
011    protected Pluralizer(@Nonnull Locale locale) {
012        super(locale);
013    }
014
015    /**
016     * Returns the plural form of the given noun.
017     */
018    @Nonnull
019    public abstract String getPlural(@Nonnull String noun);
020}
