001package org.anarres.linguistics.de;
002
003import java.util.Locale;
004import java.util.Map;
005
006import org.anarres.linguistics.core.Conjugator;
007
008/**
009 */
010// @see http://www.utils.ex.ac.uk/german/abinitio/verbs/
011public class GermanConjugator extends Conjugator<GermanConjugation> {
012
013    /*
014     * private static class Verb {
015     * private String   s1;
016     * private String   s2;
017     * private String   s3;
018     * private String   pl;
019     * public Verb(String s1, String s2, String s3, String pl) {
020     * this.s1 = s1;
021     * this.s2 = s2;
022     * this.s3 = s3;
023     * this.pl = pl;
024     * }
025     * public String getConjugation(Conjugation c) {
026     * switch ((EnglishConjugation)c.getValue()) {
027     * case I:
028     * return s1;
029     * case YOU_S:
030     * return s2;
031     * case HE:
032     * return s3;
033     * case WE:
034     * case YOU_PL:
035     * case THEY:
036     * return pl;
037     * }
038     * throw new IllegalArgumentException(
039     * "Illegal conjugation requested: " + c
040     * );
041     * }
042     * }
043     *
044     * private Map<String,Verb> irregular;
045     */
046    private Map<String, String[]> conjugations;
047
048    public GermanConjugator(Locale locale) {
049        super(locale);
050        setConjugation(GermanConjugation.values());
051
052        // irregular = new HashMap<String,Verb>();
053        conjugations.put("",
054                new String[]{"e", "st", "t", "en", "t", "en"});
055        conjugations.put("den",
056                new String[]{"e", "st", "et", "en", "et", "en"});
057        conjugations.put("eln",
058                new String[]{"e", "st", "t", "n", "t", "n"});
059        conjugations.put("ern",
060                new String[]{"e", "st", "t", "n", "t", "n"});
061        conjugations.put("ieren",
062                new String[]{"e", "st", "t", "en", "t", "en"});
063        conjugations.put("men", // only after consonant
064                new String[]{"e", "est", "et", "en", "et", "en"});
065        conjugations.put("nen", // only after consonant
066                new String[]{"e", "est", "et", "en", "et", "en"});
067        conjugations.put("sen",
068                new String[]{"e", "t", "t", "en", "t", "en"});
069        conjugations.put("ssen",
070                new String[]{"e", "t", "t", "en", "t", "en"});
071        conjugations.put("ten",
072                new String[]{"e", "est", "et", "en", "et", "en"});
073        conjugations.put("zen",
074                new String[]{"e", "t", "t", "en", "t", "en"});
075    }
076
077    @Override
078    public String getConjugation(String orig, GermanConjugation c) {
079        String word = orig.toLowerCase();
080
081        /*
082         * Verb v = irregular.get(word);
083         * if (v != null) return v.getConjugation(c);
084         */
085        for (int i = 5; i >= 0; i--) {
086            String[] conj = conjugations.get(suffix(word, i));
087            if (conj != null)
088                return conj[c.ordinal()];
089        }
090        /* The map above contains "". */
091        throw new IllegalStateException();
092    }
093
094}
