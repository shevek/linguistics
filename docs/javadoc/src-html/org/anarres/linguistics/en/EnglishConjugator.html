001package org.anarres.linguistics.en;
002
003import java.util.HashMap;
004import java.util.Locale;
005import java.util.Map;
006
007import org.anarres.linguistics.core.Conjugator;
008
009import static org.anarres.linguistics.en.EnglishConjugation.*;
010
011public class EnglishConjugator extends Conjugator<EnglishConjugation> {
012
013    private static class Verb {
014
015        private final String s1;
016        private final String s2;
017        private final String s3;
018        private final String pl;
019
020        public Verb(String s1, String s2, String s3, String pl) {
021            this.s1 = s1;
022            this.s2 = s2;
023            this.s3 = s3;
024            this.pl = pl;
025        }
026
027        public String getConjugation(EnglishConjugation c) {
028            switch (c) {
029                case I:
030                    return s1;
031                case YOU_S:
032                    return s2;
033                case HE:
034                    return s3;
035                case WE:
036                case YOU_PL:
037                case THEY:
038                    return pl;
039            }
040            throw new IllegalArgumentException(
041                    "Illegal conjugation requested: " + c
042            );
043        }
044    }
045
046    private final Map<String, Verb> irregular;
047
048    public EnglishConjugator(Locale locale) {
049        super(locale);
050        setConjugation(EnglishConjugation.values());
051
052        irregular = new HashMap<String, Verb>();
053
054        /* Fixed irregular present. */
055        for (String word : new String[]{
056            "did", "had", "ate", "made", "put",
057            "spent", "fought", "sank", "gave", "sought",
058            "shall", "could", "ought", "should",}) {
059            addFixed(word);
060        }
061
062        /* Fixed irregular present. */
063        for (String word : new String[]{
064            "thought", "saw", "bent", "will", "might", "cut",}) {
065            addFixed(word);
066        }
067
068        /* Fixed irregular non-present. */
069        addIrregular("be", "am", "are", "is", "are");
070        addIrregular("can", "can", "can", "can", "can");
071        addIrregular("cannot", "cannot", "cannot", "cannot", "cannot");
072        addIrregular("do", "do", "do", "does", "do");
073        addIrregular("go", "go", "go", "goes", "go");
074        addIrregular("have", "have", "have", "has", "have");
075        addIrregular("would", "would", "would", "would", "would");
076
077        // addIrregular("was",  "was", "were", "was", "were");
078    }
079
080    public void addIrregular(String root,
081            String s1, String s2, String s3, String pl) {
082        irregular.put(root, new Verb(s1, s2, s3, pl));
083    }
084
085    private void addFixed(String root) {
086        addIrregular(root, root, root, root, root);
087    }
088
089    @Override
090    public String getConjugation(String orig, EnglishConjugation c) {
091        String word = orig.toLowerCase();
092
093        Verb v = irregular.get(word);
094        if (v != null)
095            return v.getConjugation(c);
096
097        if (c != HE)
098            return orig;
099        switch (word.charAt(word.length() - 1)) {
100            case 'y':
101                switch (word.charAt(word.length() - 2)) {
102                    case 'a':
103                    case 'e':
104                    case 'i':
105                    case 'o':
106                    case 'u':
107                        // -ya -ey -iy -oy -uy
108                        return orig + "s";
109                    default:
110                        // -y
111                        return prefix(orig, 1) + "ies";
112                }
113            case 'h':
114                switch (word.charAt(word.length() - 2)) {
115                    case 'c':
116                    case 's':
117                        // -ch -sh
118                        return orig + "es";
119                    default:
120                        // -h
121                        return orig + "s";
122                }
123            case 'z':
124                switch (word.charAt(word.length() - 2)) {
125                    case 'a':
126                    case 'e':
127                    case 'i':
128                    case 'o':
129                    case 'u':
130                        // -az -ez -iz -oz -uz
131                        return orig + "zes";
132                    case 'z':
133                        // -zz
134                        return orig + "es";
135                    default:
136                        // -z
137                        return prefix(orig, 1) + "ies";
138                }
139            case 's':
140                // -s
141                return orig + "es";
142            default:
143                return orig + "s";
144        }
145    }
146
147}
