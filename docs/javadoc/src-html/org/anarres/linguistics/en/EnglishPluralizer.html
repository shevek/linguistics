296    public void addIrregular(String from, String to) {
297        int idx = to.indexOf('|');
298        if (idx == -1)
299            irregular.put(from, to);
300        else if (classical)
301            irregular.put(from, to.substring(idx + 1));
302        else
303            irregular.put(from, to.substring(0, idx));
304    }
305
306    public void addIrregular(String[] words, String from, String to) {
307        int flen = from.length();
308        for (String word : words) {
309            assert word.endsWith(from);
310            addIrregular(word, prefix(word, flen) + to);
311        }
312    }
313
314    public void addSuffix(String from, String to, int priority) {
315        suffixes.put(from, new Rule(to, priority));
316    }
317
318    private String getPlural(String word, String from, String to) {
319        return prefix(word, from.length()) + to;
320    }
321
322    @Override
323    public String getPlural(String orig) {
324        String word = orig.toLowerCase();
325
326        String value = irregular.get(word);
327        if (value != null)
328            return value;
329
330        int len = word.length();
331        int priority = 0;
332        int chop = len;
333        String tail = "s";
334        for (int i = 5; i > 0; i--) {
335            int start = len - i;
336            if (start < 0)
337                continue;
338            Rule rule = suffixes.get(word.substring(start));
339            if (rule != null) {
340                if (rule.priority > priority) {
341                    priority = rule.priority;
342                    tail = rule.tail;
343                    chop = start;
344                }
345            }
346        }
347
348        if (priority <= 35)
349            if (Character.isUpperCase(orig.charAt(0)))
350                if (orig.endsWith("y"))
351                    return getPlural(orig, "y", "ys");
352        if (priority <= 65)
353            if (Character.isUpperCase(orig.charAt(0)))
354                if (orig.endsWith("s"))
355                    return getPlural(orig, "s", "ses");
356        if (classical && priority <= 75) {
357            if (orig.endsWith("ynx"))
358                return getPlural(orig, "ynx", "ynges");
359            if (orig.endsWith("inx"))
360                return getPlural(orig, "inx", "inges");
361            if (orig.endsWith("anx"))
362                return getPlural(orig, "anx", "anges");
363        }
364        if (word.indexOf(' ') != -1) {
365            int idx;
366            if ((idx = word.indexOf(" of ")) != -1)
367                return getPlural(word.substring(0, idx).trim())
368                        + word.substring(idx);
369            if ((idx = word.indexOf(" in ")) != -1)
370                return getPlural(word.substring(0, idx).trim())
371                        + word.substring(idx);
372            if ((idx = word.indexOf(" to ")) != -1)
373                return getPlural(word.substring(0, idx).trim())
374                        + word.substring(idx);
375            if ((idx = word.indexOf(" at ")) != -1)
376                return getPlural(word.substring(0, idx).trim())
377                        + word.substring(idx);
378            if ((idx = word.indexOf(" de ")) != -1)
379                return getPlural(word.substring(0, idx).trim())
380                        + word.substring(idx);
381        }
382
383        return word.substring(0, chop) + tail;
384    }
